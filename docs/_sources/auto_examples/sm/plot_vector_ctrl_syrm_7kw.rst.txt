        sensorless=True,
        T_s=250e-6,
        alpha_c=2*np.pi*200, alpha_fw=2*np.pi*20, alpha_s=2*np.pi*4,

        J=.015)
    ctrl = mt.SynchronousMotorVectorCtrl(pars)
